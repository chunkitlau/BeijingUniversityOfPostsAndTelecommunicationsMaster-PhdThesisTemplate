\begin{notations}
    $(\cdot)^*$ & 复共轭\\
    $(\cdot)^{\mathrm T}$ & 矩阵转置\\
    $(\cdot)^{\mathrm H}$ & 矩阵共轭转置\\
    $\mathbf{X}$ & 矩阵或向量\\
    $\mathcal{A}$ & 集合\\
    $\mathcal{A}\times\mathcal{B}$ &
      集合 $\mathcal{A}$ 与集合 $\mathcal{B}$ 的 Cartesian 积，即 $\mathcal{A}\times\mathcal{B}=\{(a,b):a\in\mathcal{A},b\in\mathcal{B}\}$\\
\end{notations}
